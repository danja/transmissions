PREFIX mcp: <http://purl.org/stuff/mcp/>
SELECT ?graph ?id ?prompt ?output ?timestamp
WHERE {
  GRAPH ?graph {
  ?s a mcp:Interaction ;
     mcp:id ?id ;
     mcp:prompt ?prompt ;
     mcp:output ?output ;
     mcp:timestamp ?timestamp .
    FILTER regex(str(?prompt), "Semantic")
}
}
